86impl Datanode {
87    pub async fn start(&mut self) -> Result<()> {
88        info!("Starting datanode instance...");
89
90        self.start_heartbeat().await?;
91        self.wait_coordinated().await;
92
93        self.start_telemetry();
94
95        if let Some(t) = self.export_metrics_task.as_ref() {
96            t.start(None).context(StartServerSnafu)?
97        }
98
99        self.services.start_all().await.context(StartServerSnafu)
100    }
101
102    pub fn server_handlers(&self) -> &ServerHandlers {
103        &self.services
104    }
105
106    pub fn start_telemetry(&self) {
107        if let Err(e) = self.greptimedb_telemetry_task.start() {
108            warn!(e; "Failed to start telemetry task!");
109        }
110    }
111
112    pub async fn start_heartbeat(&mut self) -> Result<()> {
113        if let Some(task) = &self.heartbeat_task {
114            // Safety: The event_receiver must exist.
115            let receiver = self.region_event_receiver.take().unwrap();
116
117            task.start(receiver, self.leases_notifier.clone()).await?;
118        }
119        Ok(())
120    }
121
122    /// If `leases_notifier` exists, it waits until leases have been obtained in all regions.
123    pub async fn wait_coordinated(&mut self) {
124        if let Some(notifier) = self.leases_notifier.take() {
125            notifier.notified().await;
126        }
127    }
128
129    pub fn setup_services(&mut self, services: ServerHandlers) {
130        self.services = services;
131    }
132
133    pub async fn shutdown(&mut self) -> Result<()> {
134        self.services
135            .shutdown_all()
136            .await
137            .context(ShutdownServerSnafu)?;
138
139        let _ = self.greptimedb_telemetry_task.stop().await;
140        if let Some(heartbeat_task) = &self.heartbeat_task {
141            heartbeat_task
142                .close()
143                .map_err(BoxedError::new)
144                .context(ShutdownInstanceSnafu)?;
145        }
146        self.region_server.stop().await?;
147        Ok(())
148    }
149
150    pub fn region_server(&self) -> RegionServer {
151        self.region_server.clone()
152    }
153
154    pub fn plugins(&self) -> Plugins {
155        self.plugins.clone()
156    }

218    pub async fn build(mut self) -> Result<Datanode> {
219        let node_id = self.opts.node_id.context(MissingNodeIdSnafu)?;
220
221        let meta_client = self.meta_client.take();
222
223        // If metasrv client is provided, we will use it to control the region server.
224        // Otherwise the region server is self-controlled, meaning no heartbeat and immediately
225        // writable upon open.
226        let controlled_by_metasrv = meta_client.is_some();
227
228        // build and initialize region server
229        let (region_event_listener, region_event_receiver) = if controlled_by_metasrv {
230            let (tx, rx) = new_region_server_event_channel();
231            (Box::new(tx) as _, Some(rx))
232        } else {
233            (Box::new(NoopRegionServerEventListener) as _, None)
234        };
235
236        let cache_registry = self.cache_registry.take().context(MissingCacheSnafu)?;
237        let schema_cache: SchemaCacheRef = cache_registry.get().context(MissingCacheSnafu)?;
238        let table_id_schema_cache: TableSchemaCacheRef =
239            cache_registry.get().context(MissingCacheSnafu)?;
240
241        let schema_metadata_manager = Arc::new(SchemaMetadataManager::new(
242            table_id_schema_cache,
243            schema_cache,
244        ));
245        let region_server = self
246            .new_region_server(schema_metadata_manager, region_event_listener)
247            .await?;
248
249        // TODO(weny): Considering introducing a readonly kv_backend trait.
250        let runtime_switch_manager = RuntimeSwitchManager::new(self.kv_backend.clone());
251        let is_recovery_mode = runtime_switch_manager
252            .recovery_mode()
253            .await
254            .context(GetMetadataSnafu)?;
255        let datanode_table_manager = DatanodeTableManager::new(self.kv_backend.clone());
256        let table_values = datanode_table_manager
257            .tables(node_id)
258            .try_collect::<Vec<_>>()
259            .await
260            .context(GetMetadataSnafu)?;
261
262        let open_all_regions = open_all_regions(
263            region_server.clone(),
264            table_values,
265            !controlled_by_metasrv,
266            self.opts.init_regions_parallelism,
267            // Ignore nonexistent regions in recovery mode.
268            is_recovery_mode,
269        );
270
271        if self.opts.init_regions_in_background {
272            // Opens regions in background.
273            common_runtime::spawn_global(async move {
274                if let Err(err) = open_all_regions.await {
275                    error!(err; "Failed to open regions during the startup.");
276                }
277            });
278        } else {
279            open_all_regions.await?;
280        }
281
282        let heartbeat_task = if let Some(meta_client) = meta_client {
283            Some(
284                HeartbeatTask::try_new(
285                    &self.opts,
286                    region_server.clone(),
287                    meta_client,
288                    cache_registry,
289                    self.plugins.clone(),
290                )
291                .await?,
292            )
293        } else {
294            None
295        };
296
297        let is_standalone = heartbeat_task.is_none();
298        let greptimedb_telemetry_task = get_greptimedb_telemetry_task(
299            Some(self.opts.storage.data_home.clone()),
300            is_standalone && self.opts.enable_telemetry,
301        )
302        .await;
303
304        let leases_notifier = if self.opts.require_lease_before_startup && !is_standalone {
305            Some(Arc::new(Notify::new()))
306        } else {
307            None
308        };
309
310        let export_metrics_task =
311            ExportMetricsTask::try_new(&self.opts.export_metrics, Some(&self.plugins))
312                .context(StartServerSnafu)?;
313
314        Ok(Datanode {
315            services: ServerHandlers::default(),
316            heartbeat_task,
317            region_server,
318            greptimedb_telemetry_task,
319            region_event_receiver,
320            leases_notifier,
321            plugins: self.plugins.clone(),
322            export_metrics_task,
323        })
324    }

326    /// Builds [ObjectStoreManager] from [StorageConfig].
327    pub async fn build_object_store_manager(cfg: &StorageConfig) -> Result<ObjectStoreManagerRef> {
328        let object_store = store::new_object_store(cfg.store.clone(), &cfg.data_home).await?;
329        let default_name = cfg.store.config_name();
330        let mut object_store_manager = ObjectStoreManager::new(default_name, object_store);
331        for store in &cfg.providers {
332            object_store_manager.add(
333                store.config_name(),
334                store::new_object_store(store.clone(), &cfg.data_home).await?,
335            );
336        }
337        Ok(Arc::new(object_store_manager))
338    }

373    async fn new_region_server(
374        &mut self,
375        schema_metadata_manager: SchemaMetadataManagerRef,
376        event_listener: RegionServerEventListenerRef,
377    ) -> Result<RegionServer> {
378        let opts: &DatanodeOptions = &self.opts;
379
380        let query_engine_factory = QueryEngineFactory::new_with_plugins(
381            // query engine in datanode only executes plan with resolved table source.
382            DummyCatalogManager::arc(),
383            None,
384            None,
385            None,
386            None,
387            None,
388            false,
389            self.plugins.clone(),
390            opts.query.clone(),
391        );
392        let query_engine = query_engine_factory.query_engine();
393
394        let table_provider_factory = self
395            .table_provider_factory
396            .clone()
397            .unwrap_or_else(|| Arc::new(DummyTableProviderFactory));
398
399        let mut region_server = RegionServer::with_table_provider(
400            query_engine,
401            common_runtime::global_runtime(),
402            event_listener,
403            table_provider_factory,
404            opts.max_concurrent_queries,
405            //TODO: revaluate the hardcoded timeout on the next version of datanode concurrency limiter.
406            Duration::from_millis(100),
407            opts.grpc.flight_compression,
408        );
409
410        let object_store_manager = Self::build_object_store_manager(&opts.storage).await?;
411        let engines = self
412            .build_store_engines(
413                object_store_manager,
414                schema_metadata_manager,
415                self.plugins.clone(),
416            )
417            .await?;
418        for engine in engines {
419            region_server.register_engine(engine);
420        }
421        if let Some(topic_stats_reporter) = self.topic_stats_reporter.take() {
422            region_server.set_topic_stats_reporter(topic_stats_reporter);
423        }
424
425        Ok(region_server)
426    }

430    /// Builds [RegionEngineRef] from `store_engine` section in `opts`
431    async fn build_store_engines(
432        &mut self,
433        object_store_manager: ObjectStoreManagerRef,
434        schema_metadata_manager: SchemaMetadataManagerRef,
435        plugins: Plugins,
436    ) -> Result<Vec<RegionEngineRef>> {
437        let mut metric_engine_config = metric_engine::config::EngineConfig::default();
438        let mut mito_engine_config = MitoConfig::default();
439        let mut file_engine_config = file_engine::config::EngineConfig::default();
440
441        for engine in &self.opts.region_engine {
442            match engine {
443                RegionEngineConfig::Mito(config) => {
444                    mito_engine_config = config.clone();
445                }
446                RegionEngineConfig::File(config) => {
447                    file_engine_config = config.clone();
448                }
449                RegionEngineConfig::Metric(metric_config) => {
450                    metric_engine_config = metric_config.clone();
451                }
452            }
453        }
454
455        let mito_engine = self
456            .build_mito_engine(
457                object_store_manager.clone(),
458                mito_engine_config,
459                schema_metadata_manager.clone(),
460                plugins.clone(),
461            )
462            .await?;
463
464        let metric_engine = MetricEngine::try_new(mito_engine.clone(), metric_engine_config)
465            .context(BuildMetricEngineSnafu)?;
466
467        let file_engine = FileRegionEngine::new(
468            file_engine_config,
469            object_store_manager.default_object_store().clone(), // TODO: implement custom storage for file engine
470        );
471
472        Ok(vec![
473            Arc::new(mito_engine) as _,
474            Arc::new(metric_engine) as _,
475            Arc::new(file_engine) as _,
476        ])
477    }

479    /// Builds [MitoEngine] according to options.
480    async fn build_mito_engine(
481        &mut self,
482        object_store_manager: ObjectStoreManagerRef,
483        mut config: MitoConfig,
484        schema_metadata_manager: SchemaMetadataManagerRef,
485        plugins: Plugins,
486    ) -> Result<MitoEngine> {
487        let opts = &self.opts;
488        if opts.storage.is_object_storage() {
489            // Enable the write cache when setting object storage
490            config.enable_write_cache = true;
491            info!("Configured 'enable_write_cache=true' for mito engine.");
492        }
493
494        let mito_engine = match &opts.wal {
495            DatanodeWalConfig::RaftEngine(raft_engine_config) => {
496                let log_store =
497                    Self::build_raft_engine_log_store(&opts.storage.data_home, raft_engine_config)
498                        .await?;
499
500                let builder = MitoEngineBuilder::new(
501                    &opts.storage.data_home,
502                    config,
503                    log_store,
504                    object_store_manager,
505                    schema_metadata_manager,
506                    plugins,
507                );
508
509                #[cfg(feature = "enterprise")]
510                let builder = builder.with_extension_range_provider_factory(
511                    self.extension_range_provider_factory.take(),
512                );
513
514                builder.try_build().await.context(BuildMitoEngineSnafu)?
515            }
516            DatanodeWalConfig::Kafka(kafka_config) => {
517                if kafka_config.create_index && opts.node_id.is_none() {
518                    warn!("The WAL index creation only available in distributed mode.")
519                }
520                let global_index_collector = if kafka_config.create_index && opts.node_id.is_some()
521                {
522                    let operator = new_object_store_without_cache(
523                        &opts.storage.store,
524                        &opts.storage.data_home,
525                    )
526                    .await?;
527                    let path = default_index_file(opts.node_id.unwrap());
528                    Some(Self::build_global_index_collector(
529                        kafka_config.dump_index_interval,
530                        operator,
531                        path,
532                    ))
533                } else {
534                    None
535                };
536
537                let log_store =
538                    Self::build_kafka_log_store(kafka_config, global_index_collector).await?;
539                self.topic_stats_reporter = Some(log_store.topic_stats_reporter());
540                let builder = MitoEngineBuilder::new(
541                    &opts.storage.data_home,
542                    config,
543                    log_store,
544                    object_store_manager,
545                    schema_metadata_manager,
546                    plugins,
547                );
548
549                #[cfg(feature = "enterprise")]
550                let builder = builder.with_extension_range_provider_factory(
551                    self.extension_range_provider_factory.take(),
552                );
553
554                builder.try_build().await.context(BuildMitoEngineSnafu)?
555            }
556        };
557        Ok(mito_engine)
558    }

609/// Open all regions belong to this datanode.
610async fn open_all_regions(
611    region_server: RegionServer,
612    table_values: Vec<DatanodeTableValue>,
613    open_with_writable: bool,
614    init_regions_parallelism: usize,
615    ignore_nonexistent_region: bool,
616) -> Result<()> {
617    let mut regions = vec![];
618    #[cfg(feature = "enterprise")]
619    let mut follower_regions = vec![];
620    for table_value in table_values {
621        for region_number in table_value.regions {
622            // Augments region options with wal options if a wal options is provided.
623            let mut region_options = table_value.region_info.region_options.clone();
624            prepare_wal_options(
625                &mut region_options,
626                RegionId::new(table_value.table_id, region_number),
627                &table_value.region_info.region_wal_options,
628            );
629
630            regions.push((
631                RegionId::new(table_value.table_id, region_number),
632                table_value.region_info.engine.clone(),
633                table_value.region_info.region_storage_path.clone(),
634                region_options,
635            ));
636        }
637
638        #[cfg(feature = "enterprise")]
639        for region_number in table_value.follower_regions {
640            // Augments region options with wal options if a wal options is provided.
641            let mut region_options = table_value.region_info.region_options.clone();
642            prepare_wal_options(
643                &mut region_options,
644                RegionId::new(table_value.table_id, region_number),
645                &table_value.region_info.region_wal_options,
646            );
647
648            follower_regions.push((
649                RegionId::new(table_value.table_id, region_number),
650                table_value.region_info.engine.clone(),
651                table_value.region_info.region_storage_path.clone(),
652                region_options,
653            ));
654        }
655    }
656    let num_regions = regions.len();
657    info!("going to open {} region(s)", num_regions);
658
659    let mut region_requests = Vec::with_capacity(regions.len());
660    for (region_id, engine, store_path, options) in regions {
661        let table_dir = table_dir(&store_path, region_id.table_id());
662        region_requests.push((
663            region_id,
664            RegionOpenRequest {
665                engine,
666                table_dir,
667                path_type: PathType::Bare,
668                options,
669                skip_wal_replay: false,
670            },
671        ));
672    }
673
674    let now = Instant::now();
675    let open_regions = region_server
676        .handle_batch_open_requests(
677            init_regions_parallelism,
678            region_requests,
679            ignore_nonexistent_region,
680        )
681        .await?;
682    info!(
683        "Opened {} regions in {:?}",
684        open_regions.len(),
685        now.elapsed()
686    );
687    if !ignore_nonexistent_region {
688        ensure!(
689            open_regions.len() == num_regions,
690            error::UnexpectedSnafu {
691                violated: format!(
692                    "Expected to open {} of regions, only {} of regions has opened",
693                    num_regions,
694                    open_regions.len()
695                )
696            }
697        );
698    } else if open_regions.len() != num_regions {
699        warn!(
700            "ignore nonexistent region, expected to open {} of regions, only {} of regions has opened",
701            num_regions,
702            open_regions.len()
703        );
704    }
705
706    for region_id in open_regions {
707        if open_with_writable {
708            if let Err(e) = region_server.set_region_role(region_id, RegionRole::Leader) {
709                error!(
710                    e; "failed to convert region {region_id} to leader"
711                );
712            }
713        }
714    }
715
716    #[cfg(feature = "enterprise")]
717    if !follower_regions.is_empty() {
718        use tokio::time::Instant;
719
720        info!(
721            "going to open {} follower region(s)",
722            follower_regions.len()
723        );
724        let mut region_requests = Vec::with_capacity(follower_regions.len());
725        let num_regions = follower_regions.len();
726        for (region_id, engine, store_path, options) in follower_regions {
727            let table_dir = table_dir(&store_path, region_id.table_id());
728            region_requests.push((
729                region_id,
730                RegionOpenRequest {
731                    engine,
732                    table_dir,
733                    path_type: PathType::Bare,
734                    options,
735                    skip_wal_replay: true,
736                },
737            ));
738        }
739
740        let now = Instant::now();
741        let open_regions = region_server
742            .handle_batch_open_requests(
743                init_regions_parallelism,
744                region_requests,
745                ignore_nonexistent_region,
746            )
747            .await?;
748        info!(
749            "Opened {} follower regions in {:?}",
750            open_regions.len(),
751            now.elapsed()
752        );
753
754        if !ignore_nonexistent_region {
755            ensure!(
756                open_regions.len() == num_regions,
757                error::UnexpectedSnafu {
758                    violated: format!(
759                        "Expected to open {} of follower regions, only {} of regions has opened",
760                        num_regions,
761                        open_regions.len()
762                    )
763                }
764            );
765        } else if open_regions.len() != num_regions {
766            warn!(
767                "ignore nonexistent region, expected to open {} of follower regions, only {} of regions has opened",
768                num_regions,
769                open_regions.len()
770            );
771        }
772    }
773
774    info!("all regions are opened");
775
776    Ok(())
777}
