137        data_home: &'a str,
138        config: MitoConfig,
139        log_store: Arc<S>,
140        object_store_manager: ObjectStoreManagerRef,
141        schema_metadata_manager: SchemaMetadataManagerRef,
142        plugins: Plugins,
143    ) -> Self {
144        Self {
145            data_home,
146            config,
147            log_store,
148            object_store_manager,
149            schema_metadata_manager,
150            plugins,
151            #[cfg(feature = "enterprise")]
152            extension_range_provider_factory: None,

154    }

156    #[cfg(feature = "enterprise")]
157    #[must_use]
158    pub fn with_extension_range_provider_factory(
159        self,
160        extension_range_provider_factory: Option<BoxedExtensionRangeProviderFactory>,
161    ) -> Self {
162        Self {
163            extension_range_provider_factory,
164            ..self
165        }
166    }
167
168    pub async fn try_build(mut self) -> Result<MitoEngine> {
169        self.config.sanitize(self.data_home)?;
170
171        let config = Arc::new(self.config);
172        let workers = WorkerGroup::start(
173            config.clone(),
174            self.log_store.clone(),
175            self.object_store_manager,
176            self.schema_metadata_manager,
177            self.plugins,
178        )
179        .await?;
180        let wal_raw_entry_reader = Arc::new(LogStoreRawEntryReader::new(self.log_store));
181        let inner = EngineInner {
182            workers,
183            config,
184            wal_raw_entry_reader,
185            #[cfg(feature = "enterprise")]
186            extension_range_provider_factory: None,
187        };
188
189        #[cfg(feature = "enterprise")]
190        let inner =
191            inner.with_extension_range_provider_factory(self.extension_range_provider_factory);
192
193        Ok(MitoEngine {
194            inner: Arc::new(inner),
195        })
196    }

287    /// Edit region's metadata by [RegionEdit] directly. Use with care.
288    /// Now we only allow adding files to region (the [RegionEdit] struct can only contain a non-empty "files_to_add" field).
289    /// Other region editing intention will result in an "invalid request" error.
290    /// Also note that if a region is to be edited directly, we MUST not write data to it thereafter.
291    pub async fn edit_region(&self, region_id: RegionId, edit: RegionEdit) -> Result<()> {
292        let _timer = HANDLE_REQUEST_ELAPSED
293            .with_label_values(&["edit_region"])
294            .start_timer();
295
296        ensure!(
297            is_valid_region_edit(&edit),
298            InvalidRequestSnafu {
299                region_id,
300                reason: "invalid region edit"
301            }
302        );
303
304        let (tx, rx) = oneshot::channel();
305        let request = WorkerRequest::EditRegion(RegionEditRequest {
306            region_id,
307            edit,
308            tx,
309        });
310        self.inner
311            .workers
312            .submit_to_worker(region_id, request)
313            .await?;
314        rx.await.context(RecvSnafu)?
315    }

360    /// Find the current version's memtables and SSTs stats by region_id.
361    /// The stats must be collected in one place one time to ensure data consistency.
362    pub fn find_memtable_and_sst_stats(
363        &self,
364        region_id: RegionId,
365    ) -> Result<(Vec<MemtableStats>, Vec<FileMeta>)> {
366        let region = self
367            .find_region(region_id)
368            .context(RegionNotFoundSnafu { region_id })?;
369
370        let version = region.version();
371        let memtable_stats = version
372            .memtables
373            .list_memtables()
374            .iter()
375            .map(|x| x.stats())
376            .collect::<Vec<_>>();
377
378        let sst_stats = version
379            .ssts
380            .levels()
381            .iter()
382            .flat_map(|level| level.files().map(|x| x.meta_ref()))
383            .cloned()
384            .collect::<Vec<_>>();
385        Ok((memtable_stats, sst_stats))
386    }

396    /// Lists all SSTs from the storage layer of all regions in the engine.
397    pub fn all_ssts_from_storage(&self) -> impl Stream<Item = Result<StorageSstEntry>> {
398        let regions = self.inner.workers.all_regions();
399
400        let mut layers_distinct_table_dirs = HashMap::new();
401        for region in regions {
402            let table_dir = region.access_layer.table_dir();
403            if !layers_distinct_table_dirs.contains_key(table_dir) {
404                layers_distinct_table_dirs
405                    .insert(table_dir.to_string(), region.access_layer.clone());
406            }
407        }
408
409        stream::iter(layers_distinct_table_dirs)
410            .map(|(_, access_layer)| access_layer.storage_sst_entries())
411            .flatten()
412    }

415/// Check whether the region edit is valid. Only adding files to region is considered valid now.
416fn is_valid_region_edit(edit: &RegionEdit) -> bool {
417    !edit.files_to_add.is_empty()
418        && edit.files_to_remove.is_empty()
419        && matches!(
420            edit,
421            RegionEdit {
422                files_to_add: _,
423                files_to_remove: _,
424                compaction_time_window: None,
425                flushed_entry_id: None,
426                flushed_sequence: None,
427            }
428        )
429}

443type TopicGroupedRegionOpenRequests = HashMap<String, Vec<(RegionId, RegionOpenRequest)>>;
444
445/// Returns requests([TopicGroupedRegionOpenRequests]) grouped by topic and remaining requests.
446fn prepare_batch_open_requests(
447    requests: Vec<(RegionId, RegionOpenRequest)>,
448) -> Result<(
449    TopicGroupedRegionOpenRequests,
450    Vec<(RegionId, RegionOpenRequest)>,
451)> {
452    let mut topic_to_regions: HashMap<String, Vec<(RegionId, RegionOpenRequest)>> = HashMap::new();
453    let mut remaining_regions: Vec<(RegionId, RegionOpenRequest)> = Vec::new();
454    for (region_id, request) in requests {
455        let options = if let Some(options) = request.options.get(WAL_OPTIONS_KEY) {
456            serde_json::from_str(options).context(SerdeJsonSnafu)?
457        } else {
458            WalOptions::RaftEngine
459        };
460        match options {
461            WalOptions::Kafka(options) => {
462                topic_to_regions
463                    .entry(options.topic)
464                    .or_default()
465                    .push((region_id, request));
466            }
467            WalOptions::RaftEngine | WalOptions::Noop => {
468                remaining_regions.push((region_id, request));
469            }
470        }
471    }
472
473    Ok((topic_to_regions, remaining_regions))
474}

509    async fn open_topic_regions(
510        &self,
511        topic: String,
512        region_requests: Vec<(RegionId, RegionOpenRequest)>,
513    ) -> Result<Vec<(RegionId, Result<AffectedRows>)>> {
514        let now = Instant::now();
515        let region_ids = region_requests
516            .iter()
517            .map(|(region_id, _)| *region_id)
518            .collect::<Vec<_>>();
519        let provider = Provider::kafka_provider(topic);
520        let (distributor, entry_receivers) = build_wal_entry_distributor_and_receivers(
521            provider.clone(),
522            self.wal_raw_entry_reader.clone(),
523            &region_ids,
524            DEFAULT_ENTRY_RECEIVER_BUFFER_SIZE,
525        );
526
527        let mut responses = Vec::with_capacity(region_requests.len());
528        for ((region_id, request), entry_receiver) in
529            region_requests.into_iter().zip(entry_receivers)
530        {
531            let (request, receiver) =
532                WorkerRequest::new_open_region_request(region_id, request, Some(entry_receiver));
533            self.workers.submit_to_worker(region_id, request).await?;
534            responses.push(async move { receiver.await.context(RecvSnafu)? });
535        }
536
537        // Waits for entries distribution.
538        let distribution =
539            common_runtime::spawn_global(async move { distributor.distribute().await });
540        // Waits for worker returns.
541        let responses = join_all(responses).await;
542        distribution.await.context(JoinSnafu)??;
543
544        let num_failure = responses.iter().filter(|r| r.is_err()).count();
545        info!(
546            "Opened {} regions for topic '{}', failures: {}, elapsed: {:?}",
547            region_ids.len() - num_failure,
548            // Safety: provider is kafka provider.
549            provider.as_kafka_provider().unwrap(),
550            num_failure,
551            now.elapsed(),
552        );
553        Ok(region_ids.into_iter().zip(responses).collect())
554    }

556    async fn handle_batch_open_requests(
557        &self,
558        parallelism: usize,
559        requests: Vec<(RegionId, RegionOpenRequest)>,
560    ) -> Result<Vec<(RegionId, Result<AffectedRows>)>> {
561        let semaphore = Arc::new(Semaphore::new(parallelism));
562        let (topic_to_region_requests, remaining_region_requests) =
563            prepare_batch_open_requests(requests)?;
564        let mut responses =
565            Vec::with_capacity(topic_to_region_requests.len() + remaining_region_requests.len());
566
567        if !topic_to_region_requests.is_empty() {
568            let mut tasks = Vec::with_capacity(topic_to_region_requests.len());
569            for (topic, region_requests) in topic_to_region_requests {
570                let semaphore_moved = semaphore.clone();
571                tasks.push(async move {
572                    // Safety: semaphore must exist
573                    let _permit = semaphore_moved.acquire().await.unwrap();
574                    self.open_topic_regions(topic, region_requests).await
575                })
576            }
577            let r = try_join_all(tasks).await?;
578            responses.extend(r.into_iter().flatten());
579        }
580
581        if !remaining_region_requests.is_empty() {
582            let mut tasks = Vec::with_capacity(remaining_region_requests.len());
583            let mut region_ids = Vec::with_capacity(remaining_region_requests.len());
584            for (region_id, request) in remaining_region_requests {
585                let semaphore_moved = semaphore.clone();
586                region_ids.push(region_id);
587                tasks.push(async move {
588                    // Safety: semaphore must exist
589                    let _permit = semaphore_moved.acquire().await.unwrap();
590                    let (request, receiver) =
591                        WorkerRequest::new_open_region_request(region_id, request, None);
592
593                    self.workers.submit_to_worker(region_id, request).await?;
594
595                    receiver.await.context(RecvSnafu)?
596                })
597            }
598
599            let results = join_all(tasks).await;
600            responses.extend(region_ids.into_iter().zip(results));
601        }
602
603        Ok(responses)
604    }

626    /// Handles the scan `request` and returns a [ScanRegion].
627    fn scan_region(&self, region_id: RegionId, request: ScanRequest) -> Result<ScanRegion> {
628        let query_start = Instant::now();
629        // Reading a region doesn't need to go through the region worker thread.
630        let region = self.find_region(region_id)?;
631        let version = region.version();
632        // Get cache.
633        let cache_manager = self.workers.cache_manager();
634
635        let scan_region = ScanRegion::new(
636            version,
637            region.access_layer.clone(),
638            request,
639            CacheStrategy::EnableAll(cache_manager),
640        )
641        .with_parallel_scan_channel_size(self.config.parallel_scan_channel_size)
642        .with_max_concurrent_scan_files(self.config.max_concurrent_scan_files)
643        .with_ignore_inverted_index(self.config.inverted_index.apply_on_query.disabled())
644        .with_ignore_fulltext_index(self.config.fulltext_index.apply_on_query.disabled())
645        .with_ignore_bloom_filter(self.config.bloom_filter_index.apply_on_query.disabled())
646        .with_start_time(query_start);
647
648        #[cfg(feature = "enterprise")]
649        let scan_region = self.maybe_fill_extension_range_provider(scan_region, region);
650
651        Ok(scan_region)
652    }

654    #[cfg(feature = "enterprise")]
655    fn maybe_fill_extension_range_provider(
656        &self,
657        mut scan_region: ScanRegion,
658        region: MitoRegionRef,
659    ) -> ScanRegion {
660        if region.is_follower()
661            && let Some(factory) = self.extension_range_provider_factory.as_ref()
662        {
663            scan_region
664                .set_extension_range_provider(factory.create_extension_range_provider(region));
665        }
666        scan_region
667    }

669    /// Converts the [`RegionRole`].
670    fn set_region_role(&self, region_id: RegionId, role: RegionRole) -> Result<()> {
671        let region = self.find_region(region_id)?;
672        region.set_role(role);
673        Ok(())
674    }
675
676    /// Sets read-only for a region and ensures no more writes in the region after it returns.
677    async fn set_region_role_state_gracefully(
678        &self,
679        region_id: RegionId,
680        region_role_state: SettableRegionRoleState,
681    ) -> Result<SetRegionRoleStateResponse> {
682        // Notes: It acquires the mutable ownership to ensure no other threads,
683        // Therefore, we submit it to the worker.
684        let (request, receiver) =
685            WorkerRequest::new_set_readonly_gracefully(region_id, region_role_state);
686        self.workers.submit_to_worker(region_id, request).await?;
687
688        receiver.await.context(RecvSnafu)
689    }
690
691    async fn sync_region(
692        &self,
693        region_id: RegionId,
694        manifest_info: RegionManifestInfo,
695    ) -> Result<(ManifestVersion, bool)> {
696        ensure!(manifest_info.is_mito(), MitoManifestInfoSnafu);
697        let manifest_version = manifest_info.data_manifest_version();
698        let (request, receiver) =
699            WorkerRequest::new_sync_region_request(region_id, manifest_version);
700        self.workers.submit_to_worker(region_id, request).await?;
701
702        receiver.await.context(RecvSnafu)?
703    }
704
705    fn role(&self, region_id: RegionId) -> Option<RegionRole> {
706        self.workers.get_region(region_id).map(|region| {
707            if region.is_follower() {
708                RegionRole::Follower
709            } else {
710                RegionRole::Leader
711            }
712        })
713    }

722    #[tracing::instrument(skip_all)]
723    async fn handle_batch_open_requests(
724        &self,
725        parallelism: usize,
726        requests: Vec<(RegionId, RegionOpenRequest)>,
727    ) -> Result<BatchResponses, BoxedError> {
728        // TODO(weny): add metrics.
729        self.inner
730            .handle_batch_open_requests(parallelism, requests)
731            .await
732            .map(|responses| {
733                responses
734                    .into_iter()
735                    .map(|(region_id, response)| {
736                        (
737                            region_id,
738                            response.map(RegionResponse::new).map_err(BoxedError::new),
739                        )
740                    })
741                    .collect::<Vec<_>>()
742            })
743            .map_err(BoxedError::new)
744    }
745
746    #[tracing::instrument(skip_all)]
747    async fn handle_request(
748        &self,
749        region_id: RegionId,
750        request: RegionRequest,
751    ) -> Result<RegionResponse, BoxedError> {
752        let _timer = HANDLE_REQUEST_ELAPSED
753            .with_label_values(&[request.request_type()])
754            .start_timer();
755
756        let is_alter = matches!(request, RegionRequest::Alter(_));
757        let is_create = matches!(request, RegionRequest::Create(_));
758        let mut response = self
759            .inner
760            .handle_request(region_id, request)
761            .await
762            .map(RegionResponse::new)
763            .map_err(BoxedError::new)?;
764
765        if is_alter {
766            self.handle_alter_response(region_id, &mut response)
767                .map_err(BoxedError::new)?;
768        } else if is_create {
769            self.handle_create_response(region_id, &mut response)
770                .map_err(BoxedError::new)?;
771        }
772
773        Ok(response)
774    }

863impl MitoEngine {
864    fn handle_alter_response(
865        &self,
866        region_id: RegionId,
867        response: &mut RegionResponse,
868    ) -> Result<()> {
869        if let Some(statistic) = self.region_statistic(region_id) {
870            Self::encode_manifest_info_to_extensions(
871                &region_id,
872                statistic.manifest,
873                &mut response.extensions,
874            )?;
875        }
876        let column_metadatas = self
877            .inner
878            .find_region(region_id)
879            .ok()
880            .map(|r| r.metadata().column_metadatas.clone());
881        if let Some(column_metadatas) = column_metadatas {
882            Self::encode_column_metadatas_to_extensions(
883                &region_id,
884                column_metadatas,
885                &mut response.extensions,
886            )?;
887        }
888        Ok(())
889    }
890
891    fn handle_create_response(
892        &self,
893        region_id: RegionId,
894        response: &mut RegionResponse,
895    ) -> Result<()> {
896        let column_metadatas = self
897            .inner
898            .find_region(region_id)
899            .ok()
900            .map(|r| r.metadata().column_metadatas.clone());
901        if let Some(column_metadatas) = column_metadatas {
902            Self::encode_column_metadatas_to_extensions(
903                &region_id,
904                column_metadatas,
905                &mut response.extensions,
906            )?;
907        }
908        Ok(())
909    }
