80impl TryFrom<BulkWalEntry> for BulkPart {
81    type Error = error::Error;
82
83    fn try_from(value: BulkWalEntry) -> std::result::Result<Self, Self::Error> {
84        match value.body.expect("Entry payload should be present") {
85            Body::ArrowIpc(ipc) => {
86                let mut decoder = FlightDecoder::try_from_schema_bytes(&ipc.schema)
87                    .context(error::ConvertBulkWalEntrySnafu)?;
88                let batch = decoder
89                    .try_decode_record_batch(&ipc.data_header, &ipc.payload)
90                    .context(error::ConvertBulkWalEntrySnafu)?;
91                Ok(Self {
92                    batch,
93                    max_ts: value.max_ts,
94                    min_ts: value.min_ts,
95                    sequence: value.sequence,
96                    timestamp_index: value.timestamp_index as usize,
97                    raw_data: Some(ipc),
98                })
99            }
100        }
101    }
102}

104impl TryFrom<&BulkPart> for BulkWalEntry {
105    type Error = error::Error;
106
107    fn try_from(value: &BulkPart) -> Result<Self> {
108        if let Some(ipc) = &value.raw_data {
109            Ok(BulkWalEntry {
110                sequence: value.sequence,
111                max_ts: value.max_ts,
112                min_ts: value.min_ts,
113                timestamp_index: value.timestamp_index as u32,
114                body: Some(Body::ArrowIpc(ipc.clone())),
115            })
116        } else {
117            let mut encoder = FlightEncoder::default();
118            let schema_bytes = encoder
119                .encode_schema(value.batch.schema().as_ref())
120                .data_header;
121            let [rb_data] = encoder
122                .encode(FlightMessage::RecordBatch(value.batch.clone()))
123                .try_into()
124                .map_err(|_| {
125                    error::UnsupportedOperationSnafu {
126                        err_msg: "create BulkWalEntry from RecordBatch with dictionary arrays",
127                    }
128                    .build()
129                })?;
130            Ok(BulkWalEntry {
131                sequence: value.sequence,
132                max_ts: value.max_ts,
133                min_ts: value.min_ts,
134                timestamp_index: value.timestamp_index as u32,
135                body: Some(Body::ArrowIpc(ArrowIpc {
136                    schema: schema_bytes,
137                    data_header: rb_data.data_header,
138                    payload: rb_data.data_body,
139                })),
140            })
141        }
142    }

146    pub(crate) fn estimated_size(&self) -> usize {
147        self.batch
148            .columns()
149            .iter()

151            .map(|c| c.to_data().get_slice_memory_size().unwrap_or(0))

155    /// Converts [BulkPart] to [Mutation] for fallback `write_bulk` implementation.
156    pub(crate) fn to_mutation(&self, region_metadata: &RegionMetadataRef) -> Result<Mutation> {
157        let vectors = region_metadata
158            .schema
159            .column_schemas()
160            .iter()
161            .map(|col| match self.batch.column_by_name(&col.name) {
162                None => Ok(None),
163                Some(col) => Helper::try_into_vector(col).map(Some),
164            })
165            .collect::<datatypes::error::Result<Vec<_>>>()
166            .context(error::ComputeVectorSnafu)?;
167
168        let rows = (0..self.num_rows())
169            .map(|row_idx| {
170                let values = (0..self.batch.num_columns())
171                    .map(|col_idx| {
172                        if let Some(v) = &vectors[col_idx] {
173                            value_to_grpc_value(v.get(row_idx))
174                        } else {
175                            api::v1::Value { value_data: None }
176                        }
177                    })
178                    .collect::<Vec<_>>();
179                api::v1::Row { values }
180            })
181            .collect::<Vec<_>>();
182
183        let schema = region_metadata
184            .column_metadatas
185            .iter()
186            .map(|c| {
187                let data_type_wrapper =
188                    ColumnDataTypeWrapper::try_from(c.column_schema.data_type.clone())?;
189                Ok(api::v1::ColumnSchema {
190                    column_name: c.column_schema.name.clone(),
191                    datatype: data_type_wrapper.datatype() as i32,
192                    semantic_type: c.semantic_type as i32,
193                    ..Default::default()
194                })
195            })
196            .collect::<api::error::Result<Vec<_>>>()
197            .context(error::ConvertColumnDataTypeSnafu {
198                reason: "failed to convert region metadata to column schema",
199            })?;
200
201        let rows = api::v1::Rows { schema, rows };
202
203        Ok(Mutation {
204            op_type: OpType::Put as i32,
205            sequence: self.sequence,
206            rows: Some(rows),
207            write_hint: None,
208        })
209    }

220/// Primary key column builder for handling strings specially.
221enum PrimaryKeyColumnBuilder {
222    /// String dictionary builder for string types.
223    StringDict(StringDictionaryBuilder<UInt32Type>),
224    /// Generic mutable vector for other types.
225    Vector(Box<dyn MutableVector>),
226}
227
228impl PrimaryKeyColumnBuilder {
229    /// Appends a value to the builder.
230    fn push_value_ref(&mut self, value: ValueRef) -> Result<()> {
231        match self {
232            PrimaryKeyColumnBuilder::StringDict(builder) => {
233                if let Some(s) = value.as_string().context(DataTypeMismatchSnafu)? {
234                    // We know the value is a string.
235                    builder.append_value(s);
236                } else {
237                    builder.append_null();
238                }
239            }
240            PrimaryKeyColumnBuilder::Vector(builder) => {
241                builder.push_value_ref(value);
242            }
243        }
244        Ok(())
245    }
246
247    /// Converts the builder to an ArrayRef.
248    fn into_arrow_array(self) -> ArrayRef {
249        match self {
250            PrimaryKeyColumnBuilder::StringDict(mut builder) => Arc::new(builder.finish()),
251            PrimaryKeyColumnBuilder::Vector(mut builder) => builder.to_vector().to_arrow_array(),
252        }
253    }
254}

272    primary_key_column_builders: Vec<PrimaryKeyColumnBuilder>,

282impl BulkPartConverter {
283    /// Creates a new converter.
284    ///
285    /// If `store_primary_key_columns` is true and the encoding is not sparse encoding, it
286    /// stores primary key columns in arrays additionally.
287    pub fn new(
288        region_metadata: &RegionMetadataRef,
289        schema: SchemaRef,
290        capacity: usize,
291        primary_key_codec: Arc<dyn PrimaryKeyCodec>,
292        store_primary_key_columns: bool,
293    ) -> Self {
294        debug_assert_eq!(
295            region_metadata.primary_key_encoding,
296            primary_key_codec.encoding()
297        );
298
299        let primary_key_column_builders = if store_primary_key_columns
300            && region_metadata.primary_key_encoding != PrimaryKeyEncoding::Sparse
301        {
302            new_primary_key_column_builders(region_metadata, capacity)
303        } else {
304            Vec::new()
305        };
306
307        Self {
308            region_metadata: region_metadata.clone(),
309            schema,
310            primary_key_codec,
311            key_buf: Vec::new(),
312            key_array_builder: PrimaryKeyArrayBuilder::new(),
313            value_builder: ValueBuilder::new(region_metadata, capacity),
314            primary_key_column_builders,
315            min_ts: i64::MAX,
316            max_ts: i64::MIN,
317            max_sequence: SequenceNumber::MIN,
318        }
319    }
320
321    /// Appends a [KeyValues] into the converter.
322    pub fn append_key_values(&mut self, key_values: &KeyValues) -> Result<()> {
323        for kv in key_values.iter() {
324            self.append_key_value(&kv)?;
325        }
326
327        Ok(())
328    }
329
330    /// Appends a [KeyValue] to builders.
331    ///
332    /// If the primary key uses sparse encoding, callers must encoded the primary key in the [KeyValue].
333    fn append_key_value(&mut self, kv: &KeyValue) -> Result<()> {
334        // Handles primary key based on encoding type
335        if self.primary_key_codec.encoding() == PrimaryKeyEncoding::Sparse {
336            // For sparse encoding, the primary key is already encoded in the KeyValue
337            // Gets the first (and only) primary key value which contains the encoded key
338            let mut primary_keys = kv.primary_keys();
339            if let Some(encoded) = primary_keys
340                .next()
341                .context(ColumnNotFoundSnafu {
342                    column: PRIMARY_KEY_COLUMN_NAME,
343                })?
344                .as_binary()
345                .context(DataTypeMismatchSnafu)?
346            {
347                self.key_array_builder
348                    .append(encoded)
349                    .context(ComputeArrowSnafu)?;
350            } else {
351                self.key_array_builder
352                    .append("")
353                    .context(ComputeArrowSnafu)?;
354            }
355        } else {
356            // For dense encoding, we need to encode the primary key columns
357            self.key_buf.clear();
358            self.primary_key_codec
359                .encode_key_value(kv, &mut self.key_buf)
360                .context(EncodeSnafu)?;
361            self.key_array_builder
362                .append(&self.key_buf)
363                .context(ComputeArrowSnafu)?;
364        };
365
366        // If storing primary key columns, append values to individual builders
367        if !self.primary_key_column_builders.is_empty() {
368            for (builder, pk_value) in self
369                .primary_key_column_builders
370                .iter_mut()
371                .zip(kv.primary_keys())
372            {
373                builder.push_value_ref(pk_value)?;
374            }
375        }
376
377        // Pushes other columns.
378        self.value_builder.push(
379            kv.timestamp(),
380            kv.sequence(),
381            kv.op_type() as u8,
382            kv.fields(),
383        );
384
385        // Updates statistics
386        // Safety: timestamp of kv must be both present and a valid timestamp value.
387        let ts = kv.timestamp().as_timestamp().unwrap().unwrap().value();
388        self.min_ts = self.min_ts.min(ts);
389        self.max_ts = self.max_ts.max(ts);
390        self.max_sequence = self.max_sequence.max(kv.sequence());
391
392        Ok(())
393    }

395    /// Converts buffered content into a [BulkPart].
396    ///
397    /// It sorts the record batch by (primary key, timestamp, sequence desc).
398    pub fn convert(mut self) -> Result<BulkPart> {
399        let values = Values::from(self.value_builder);
400        let mut columns =
401            Vec::with_capacity(4 + values.fields.len() + self.primary_key_column_builders.len());
402
403        // Build primary key column arrays if enabled.
404        for builder in self.primary_key_column_builders {
405            columns.push(builder.into_arrow_array());
406        }
407        // Then fields columns.
408        columns.extend(values.fields.iter().map(|field| field.to_arrow_array()));
409        // Time index.
410        let timestamp_index = columns.len();
411        columns.push(values.timestamp.to_arrow_array());
412        // Primary key.
413        let pk_array = self.key_array_builder.finish();
414        columns.push(Arc::new(pk_array));
415        // Sequence and op type.
416        columns.push(values.sequence.to_arrow_array());
417        columns.push(values.op_type.to_arrow_array());
418
419        let batch = RecordBatch::try_new(self.schema, columns).context(NewRecordBatchSnafu)?;
420        // Sorts the record batch.
421        let batch = sort_primary_key_record_batch(&batch)?;
422
423        Ok(BulkPart {
424            batch,
425            max_ts: self.max_ts,
426            min_ts: self.min_ts,
427            sequence: self.max_sequence,
428            timestamp_index,
429            raw_data: None,
430        })
431    }
432}

434fn new_primary_key_column_builders(
435    metadata: &RegionMetadata,
436    capacity: usize,
437) -> Vec<PrimaryKeyColumnBuilder> {
438    metadata
439        .primary_key_columns()
440        .map(|col| {
441            if col.column_schema.data_type.is_string() {
442                PrimaryKeyColumnBuilder::StringDict(StringDictionaryBuilder::with_capacity(
443                    capacity,
444                    INIT_DICT_VALUE_CAPACITY,
445                    capacity,
446                ))
447            } else {
448                PrimaryKeyColumnBuilder::Vector(
449                    col.column_schema.data_type.create_mutable_vector(capacity),
450                )
451            }
452        })
453        .collect()
454}

456/// Sorts the record batch with primary key format.
457fn sort_primary_key_record_batch(batch: &RecordBatch) -> Result<RecordBatch> {
458    let total_columns = batch.num_columns();
459    let sort_columns = vec![
460        // Primary key column (ascending)
461        SortColumn {
462            values: batch.column(total_columns - 3).clone(),
463            options: Some(SortOptions {
464                descending: false,
465                nulls_first: true,
466            }),
467        },
468        // Time index column (ascending)
469        SortColumn {
470            values: batch.column(total_columns - 4).clone(),
471            options: Some(SortOptions {
472                descending: false,
473                nulls_first: true,
474            }),
475        },
476        // Sequence column (descending)
477        SortColumn {
478            values: batch.column(total_columns - 2).clone(),
479            options: Some(SortOptions {
480                descending: true,
481                nulls_first: true,
482            }),
483        },
484    ];
485
486    let indices = datatypes::arrow::compute::lexsort_to_indices(&sort_columns, None)
487        .context(ComputeArrowSnafu)?;
488
489    datatypes::arrow::compute::take_record_batch(batch, &indices).context(ComputeArrowSnafu)
490}

507    pub(crate) fn read(
508        &self,
509        context: BulkIterContextRef,
510        sequence: Option<SequenceNumber>,
511    ) -> Result<Option<BoxedRecordBatchIterator>> {
512        // use predicate to find row groups to read.
513        let row_groups_to_read = context.row_groups_to_read(&self.metadata.parquet_metadata);
514
515        if row_groups_to_read.is_empty() {
516            // All row groups are filtered.
517            return Ok(None);
518        }
519
520        let iter = EncodedBulkPartIter::try_new(
521            context,
522            row_groups_to_read,
523            self.metadata.parquet_metadata.clone(),
524            self.data.clone(),
525            sequence,
526        )?;
527        Ok(Some(Box::new(iter) as BoxedRecordBatchIterator))
528    }
529}

567impl BulkPartEncoder {
568    /// Encodes bulk part to a [EncodedBulkPart], returns the encoded data.
569    fn encode_part(&self, part: &BulkPart) -> Result<Option<EncodedBulkPart>> {
570        if part.batch.num_rows() == 0 {
571            return Ok(None);
572        }
573
574        let mut buf = Vec::with_capacity(4096);
575        let arrow_schema = part.batch.schema();
576
577        let file_metadata = {
578            let mut writer =
579                ArrowWriter::try_new(&mut buf, arrow_schema, self.writer_props.clone())
580                    .context(EncodeMemtableSnafu)?;
581            writer.write(&part.batch).context(EncodeMemtableSnafu)?;
582            writer.finish().context(EncodeMemtableSnafu)?
583        };
584
585        let buf = Bytes::from(buf);
586        let parquet_metadata = Arc::new(parse_parquet_metadata(file_metadata)?);
587
588        Ok(Some(EncodedBulkPart {
589            data: buf,
590            metadata: BulkPartMeta {
591                num_rows: part.batch.num_rows(),
592                max_timestamp: part.max_ts,
593                min_timestamp: part.min_ts,
594                parquet_metadata,
595                region_metadata: self.metadata.clone(),
596            },
597        }))
598    }

601/// Converts mutations to record batches.
602fn mutations_to_record_batch(
603    mutations: &[Mutation],
604    metadata: &RegionMetadataRef,
605    pk_encoder: &DensePrimaryKeyCodec,
606    dedup: bool,
607) -> Result<Option<(RecordBatch, i64, i64)>> {
608    let total_rows: usize = mutations
609        .iter()
610        .map(|m| m.rows.as_ref().map(|r| r.rows.len()).unwrap_or(0))
611        .sum();
612
613    if total_rows == 0 {
614        return Ok(None);
615    }
616
617    let mut pk_builder = BinaryBuilder::with_capacity(total_rows, 0);
618
619    let mut ts_vector: Box<dyn MutableVector> = metadata
620        .time_index_column()
621        .column_schema
622        .data_type
623        .create_mutable_vector(total_rows);
624    let mut sequence_builder = UInt64Builder::with_capacity(total_rows);
625    let mut op_type_builder = UInt8Builder::with_capacity(total_rows);
626
627    let mut field_builders: Vec<Box<dyn MutableVector>> = metadata
628        .field_columns()
629        .map(|f| f.column_schema.data_type.create_mutable_vector(total_rows))
630        .collect();
631
632    let mut pk_buffer = vec![];
633    for m in mutations {
634        let Some(key_values) = KeyValuesRef::new(metadata, m) else {
635            continue;
636        };
637
638        for row in key_values.iter() {
639            pk_buffer.clear();
640            pk_encoder
641                .encode_to_vec(row.primary_keys(), &mut pk_buffer)
642                .context(EncodeSnafu)?;
643            pk_builder.append_value(pk_buffer.as_bytes());
644            ts_vector.push_value_ref(row.timestamp());
645            sequence_builder.append_value(row.sequence());
646            op_type_builder.append_value(row.op_type() as u8);
647            for (builder, field) in field_builders.iter_mut().zip(row.fields()) {
648                builder.push_value_ref(field);
649            }
650        }
651    }
652
653    let arrow_schema = to_sst_arrow_schema(metadata);
654    // safety: timestamp column must be valid, and values must not be None.
655    let timestamp_unit = metadata
656        .time_index_column()
657        .column_schema
658        .data_type
659        .as_timestamp()
660        .unwrap()
661        .unit();
662    let sorter = ArraysSorter {
663        encoded_primary_keys: pk_builder.finish(),
664        timestamp_unit,
665        timestamp: ts_vector.to_vector().to_arrow_array(),
666        sequence: sequence_builder.finish(),
667        op_type: op_type_builder.finish(),
668        fields: field_builders
669            .iter_mut()
670            .map(|f| f.to_vector().to_arrow_array()),
671        dedup,
672        arrow_schema,
673    };
674
675    sorter.sort().map(Some)
676}

690where
691    I: Iterator<Item = ArrayRef>,
692{
693    /// Converts arrays to record batch.
694    fn sort(self) -> Result<(RecordBatch, i64, i64)> {
695        debug_assert!(!self.timestamp.is_empty());
696        debug_assert!(self.timestamp.len() == self.sequence.len());
697        debug_assert!(self.timestamp.len() == self.op_type.len());
698        debug_assert!(self.timestamp.len() == self.encoded_primary_keys.len());
699
700        let timestamp_iter = timestamp_array_to_iter(self.timestamp_unit, &self.timestamp);
701        let (mut min_timestamp, mut max_timestamp) = (i64::MAX, i64::MIN);
702        let mut to_sort = self
703            .encoded_primary_keys
704            .iter()
705            .zip(timestamp_iter)
706            .zip(self.sequence.iter())
707            .map(|((pk, timestamp), sequence)| {
708                max_timestamp = max_timestamp.max(*timestamp);
709                min_timestamp = min_timestamp.min(*timestamp);
710                (pk, timestamp, sequence)
711            })
712            .enumerate()
713            .collect::<Vec<_>>();
714
715        to_sort.sort_unstable_by(|(_, (l_pk, l_ts, l_seq)), (_, (r_pk, r_ts, r_seq))| {
716            l_pk.cmp(r_pk)
717                .then(l_ts.cmp(r_ts))
718                .then(l_seq.cmp(r_seq).reverse())
719        });
720
721        if self.dedup {
722            // Dedup by timestamps while ignore sequence.
723            to_sort.dedup_by(|(_, (l_pk, l_ts, _)), (_, (r_pk, r_ts, _))| {
724                l_pk == r_pk && l_ts == r_ts
725            });
726        }
727
728        let indices = UInt32Array::from_iter_values(to_sort.iter().map(|v| v.0 as u32));
729
730        let pk_dictionary = Arc::new(binary_array_to_dictionary(
731            // safety: pk must be BinaryArray
732            arrow::compute::take(
733                &self.encoded_primary_keys,
734                &indices,
735                Some(TakeOptions {
736                    check_bounds: false,
737                }),
738            )
739            .context(ComputeArrowSnafu)?
740            .as_any()
741            .downcast_ref::<BinaryArray>()
742            .unwrap(),
743        )?) as ArrayRef;
744
745        let mut arrays = Vec::with_capacity(self.arrow_schema.fields.len());
746        for arr in self.fields {
747            arrays.push(
748                arrow::compute::take(
749                    &arr,
750                    &indices,
751                    Some(TakeOptions {
752                        check_bounds: false,
753                    }),
754                )
755                .context(ComputeArrowSnafu)?,
756            );
757        }
758
759        let timestamp = arrow::compute::take(
760            &self.timestamp,
761            &indices,
762            Some(TakeOptions {
763                check_bounds: false,
764            }),
765        )
766        .context(ComputeArrowSnafu)?;
767
768        arrays.push(timestamp);
769        arrays.push(pk_dictionary);
770        arrays.push(
771            arrow::compute::take(
772                &self.sequence,
773                &indices,
774                Some(TakeOptions {
775                    check_bounds: false,
776                }),
777            )
778            .context(ComputeArrowSnafu)?,
779        );
780
781        arrays.push(
782            arrow::compute::take(
783                &self.op_type,
784                &indices,
785                Some(TakeOptions {
786                    check_bounds: false,
787                }),
788            )
789            .context(ComputeArrowSnafu)?,
790        );
791
792        let batch = RecordBatch::try_new(self.arrow_schema, arrays).context(NewRecordBatchSnafu)?;
793        Ok((batch, min_timestamp, max_timestamp))
794    }
795}

798fn timestamp_array_to_iter(
799    timestamp_unit: TimeUnit,
800    timestamp: &ArrayRef,
801) -> impl Iterator<Item = &i64> {
802    match timestamp_unit {
803        // safety: timestamp column must be valid.
804        TimeUnit::Second => timestamp
805            .as_any()
806            .downcast_ref::<TimestampSecondArray>()
807            .unwrap()
808            .values()
809            .iter(),

811            .as_any()
812            .downcast_ref::<TimestampMillisecondArray>()
813            .unwrap()

816        TimeUnit::Microsecond => timestamp
817            .as_any()

819            .unwrap()
820            .values()
821            .iter(),
822        TimeUnit::Nanosecond => timestamp
823            .as_any()
824            .downcast_ref::<TimestampNanosecondArray>()
825            .unwrap()
826            .values()
827            .iter(),
828    }

831/// Converts a **sorted** [BinaryArray] to [DictionaryArray].
832fn binary_array_to_dictionary(input: &BinaryArray) -> Result<PrimaryKeyArray> {
833    if input.is_empty() {
834        return Ok(DictionaryArray::new(
835            UInt32Array::from(Vec::<u32>::new()),
836            Arc::new(BinaryArray::from_vec(vec![])) as ArrayRef,
837        ));
838    }
839    let mut keys = Vec::with_capacity(16);
840    let mut values = BinaryBuilder::new();
841    let mut prev: usize = 0;
842    keys.push(prev as u32);
843    values.append_value(input.value(prev));
844
845    for current_bytes in input.iter().skip(1) {
846        // safety: encoded pk must present.
847        let current_bytes = current_bytes.unwrap();
848        let prev_bytes = input.value(prev);
849        if current_bytes != prev_bytes {
850            values.append_value(current_bytes);
851            prev += 1;
852        }
853        keys.push(prev as u32);
854    }
855
856    Ok(DictionaryArray::new(
857        UInt32Array::from(keys),
858        Arc::new(values.finish()) as ArrayRef,
859    ))
860}

1602    fn build_key_values_with_sparse_encoding(
1603        metadata: &RegionMetadataRef,
1604        primary_key_codec: &Arc<dyn PrimaryKeyCodec>,
1605        table_id: u32,
1606        tsid: u64,
1607        k0: String,
1608        k1: String,
1609        timestamps: impl Iterator<Item = i64>,
1610        values: impl Iterator<Item = Option<f64>>,
1611        sequence: SequenceNumber,
1612    ) -> KeyValues {
1613        // Encode the primary key (__table_id, __tsid, k0, k1) into binary format using the sparse codec
1614        let pk_values = vec![
1615            (ReservedColumnId::table_id(), Value::UInt32(table_id)),
1616            (ReservedColumnId::tsid(), Value::UInt64(tsid)),
1617            (0, Value::String(k0.clone().into())),
1618            (1, Value::String(k1.clone().into())),
1619        ];
1620        let mut encoded_key = Vec::new();
1621        primary_key_codec
1622            .encode_values(&pk_values, &mut encoded_key)
1623            .unwrap();
1624        assert!(!encoded_key.is_empty());
1625
1626        // Create schema for sparse encoding: __primary_key, ts, v0, v1
1627        let column_schema = vec![
1628            api::v1::ColumnSchema {
1629                column_name: PRIMARY_KEY_COLUMN_NAME.to_string(),
1630                datatype: api::helper::ColumnDataTypeWrapper::try_from(
1631                    ConcreteDataType::binary_datatype(),
1632                )
1633                .unwrap()
1634                .datatype() as i32,
1635                semantic_type: api::v1::SemanticType::Tag as i32,
1636                ..Default::default()
1637            },
1638            api::v1::ColumnSchema {
1639                column_name: "ts".to_string(),
1640                datatype: api::helper::ColumnDataTypeWrapper::try_from(
1641                    ConcreteDataType::timestamp_millisecond_datatype(),
1642                )
1643                .unwrap()
1644                .datatype() as i32,
1645                semantic_type: api::v1::SemanticType::Timestamp as i32,
1646                ..Default::default()
1647            },
1648            api::v1::ColumnSchema {
1649                column_name: "v0".to_string(),
1650                datatype: api::helper::ColumnDataTypeWrapper::try_from(
1651                    ConcreteDataType::int64_datatype(),
1652                )
1653                .unwrap()
1654                .datatype() as i32,
1655                semantic_type: api::v1::SemanticType::Field as i32,
1656                ..Default::default()
1657            },
1658            api::v1::ColumnSchema {
1659                column_name: "v1".to_string(),
1660                datatype: api::helper::ColumnDataTypeWrapper::try_from(
1661                    ConcreteDataType::float64_datatype(),
1662                )
1663                .unwrap()
1664                .datatype() as i32,
1665                semantic_type: api::v1::SemanticType::Field as i32,
1666                ..Default::default()
1667            },
1668        ];
1669
1670        let rows = timestamps
1671            .zip(values)
1672            .map(|(ts, v)| Row {
1673                values: vec![
1674                    api::v1::Value {
1675                        value_data: Some(api::v1::value::ValueData::BinaryValue(
1676                            encoded_key.clone(),
1677                        )),
1678                    },
1679                    api::v1::Value {
1680                        value_data: Some(api::v1::value::ValueData::TimestampMillisecondValue(ts)),
1681                    },
1682                    api::v1::Value {
1683                        value_data: Some(api::v1::value::ValueData::I64Value(ts)),
1684                    },
1685                    api::v1::Value {
1686                        value_data: v.map(api::v1::value::ValueData::F64Value),
1687                    },
1688                ],
1689            })
1690            .collect();
1691
1692        let mutation = api::v1::Mutation {
1693            op_type: 1,
1694            sequence,
1695            rows: Some(api::v1::Rows {
1696                schema: column_schema,
1697                rows,
1698            }),
1699            write_hint: Some(WriteHint {
1700                primary_key_encoding: api::v1::PrimaryKeyEncoding::Sparse.into(),
1701            }),
1702        };
1703        KeyValues::new(metadata.as_ref(), mutation).unwrap()
1704    }
